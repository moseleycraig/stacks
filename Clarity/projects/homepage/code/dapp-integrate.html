Sample API Integration

// Guardian Gateway API Integration Example import { GuardianGateway } from '@guardian/sdk'; const gg = new GuardianGateway({ apiKey: 'your-api-key', network: 'mainnet' }); // Get current price with confidence const btcPrice = await gg.getPrice('BTC'); console.log(`BTC: ${btcPrice.price} (${btcPrice.confidence}% confidence)`); // Subscribe to real-time updates gg.subscribe('BTC', (update) => { console.log('Price update:', update); });
